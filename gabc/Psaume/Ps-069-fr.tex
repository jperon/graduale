%% Ps. 69
\versus{Mon Dieu, venez à mon aide ; Seigneur, hâtez-vous de me secourir !}
\versus{Qu'ils soient confondus et repoussés, ceux qui en veulent à ma vie.}
\versus{Qu'ils reculent confondus, ceux qui me veulent du mal.}
\versus{Qu'ils reculent subitement dans la confusion, ceux qui me disent : Va ! Va !}
\versus{Mais qu'ils se réjouissent et exultent en vous, tous ceux qui vous cherchent, et qu'ils disent toujours : \og Dieu soit loué\fg, ceux qui aspirent à votre salut.}
\versus{Quant à moi, je suis pauvre et démuni ; Seigneur, secourez-moi !}
