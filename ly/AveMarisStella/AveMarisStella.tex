\couplet{
	Ave maris Stella,\\*
	Dei Mater alma,\\*
	Atque semper virgo,\\*
	Felix cæli porta.
}

\couplet{
	Sumens illud Ave\\*
	Gabrielis ore,\\
	Funda nos in pace,\\*
	Mutans Evæ nomen.
}

\couplet{
	Solve vincla reis,\\*
	Profer lumen cæcis ;\\
	Mala nostra pelle,\\*
	Bona cuncta posce.
}

\couplet{
	Montra t\emph{e} esse matrem,\\*
	Sumat per te preces\\
	Qui, pro nobis natus,\\*
	Tulit esse tuus.
}

\couplet{
	Virgo singularis,\\*
	Inter omnes mitis,\\
	Nos culpis solutos,\\*
	Mites fac et castos.
}

\couplet{
	Vitam præsta puram,\\*
	Iter para tutum,\\
	Ut videntes Iesum,\\*
	Semper collætemur.
}

\couplet{
	Sit laus Deo Patri,\\*
	Summo Christo decus ;\\*
	Spiritui Sancto,\\*
	Tribus honor unus.
	Amen.
}
