\refrain{
	Lauda, Ierusalem Dominum :\\*
	lauda Deum tuum, Sion.\\*
	Hosanna, Hosanna, Hosanna Filio David.
}

\couplet{
	Quoniam confortavit seras portarum tuarum :
	benedixit filiis tuis in te.
}

\couplet{
	Qui posuit fines tuos pacem :
	et adipe frumenti satiat te.
}

\couplet{
	Qui emittit eloquium suum terræ :
	velociter currit sermo eius. 
}

\couplet{
	Qui dat nivem sicut lanam :
	nebulam sicut cinerem spargit.
}

\couplet{
	Mittit crystallum suam sicut buccellas :
	ante faciem frigoris eius quis sustinébit ?
}

\couplet{
	Emittet verbum suum, et liquefaciet ea :
	flabit spiritus eius, et fluent aquæ. 
}

\couplet{
	Qui annuntiat verbum suum Iacob :
	iustitias et iudicia sua Israel.
}

\couplet{
	Non fecit taliter omni nationi :
	et iudicia sua non manifestavit eis.
}

\couplet{
	Gloria Patri, et Filio,
	et Spiritui Sancto.
}

\couplet{
	Sicut erat in principio, et nunc, et semper,
	et in sæcula sæculorum. Amen.
}
