\refrain{
Ô ma reine, ô Vierge Marie,\\*
Je vous donne mon cœur ;\\*
Je vous consacre pour la vie\\*
Mes peines, mon bonheur.
}

\couplet{
Je me donne à vous, tendre Mère,\\*
Je me jette en vos bras ;\\
\bis{
	Marie, exaucez ma prière,\\*
	Ne m’abandonnez pas.
}
}

\couplet{
Je vous offre mon corps, mon âme,\\*
Aujourd’hui pour toujours,\\
Marie, et de vous je réclame\\*
Un maternel secours.
}

\couplet{
Je vous offre mes espérances,\\*
Et ma joie et mes pleurs ;\\
Marie, à vos propres souffrances\\*
J’unirai mes douleurs.
}

\couplet{
Je vous offre toute ma vie,\\*
Soumis à votre loi ;\\
Marie, au jour de l’agonie,\\*
Soyez auprès de moi.
}
